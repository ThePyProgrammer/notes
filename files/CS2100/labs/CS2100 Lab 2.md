title: Code to Produce Output

int hexToDecimal(char hex[], size_t size) {
	if(size == 0) return 0;
    int o = 0, exp = 1;
    char * ptr = &(hex[size-1]);
    do {
        o += hexVal(*ptr) * exp;
        exp *= 16;
    } while((ptr--) > &hex);
    return o;
}
